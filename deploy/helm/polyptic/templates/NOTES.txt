1. Reach the control plane
{{- if .Values.ingress.enabled }}

   Ingress is enabled:
     http{{ if .Values.ingress.tls.enabled }}s{{ end }}://{{ .Values.ingress.host }}{{ .Values.ingress.path }}
{{- else if eq .Values.service.type "NodePort" }}

   NodePort service. Find the port + a node IP:
     export NODE_PORT=$(kubectl get svc {{ include "polyptic.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}")
     export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
     echo "http://$NODE_IP:$NODE_PORT"
{{- else if eq .Values.service.type "LoadBalancer" }}

   LoadBalancer service. Wait for the external IP:
     kubectl get svc {{ include "polyptic.fullname" . }} -n {{ .Release.Namespace }} -w
{{- else }}

   ClusterIP service (in-cluster only). Port-forward for local access:
     kubectl port-forward svc/{{ include "polyptic.fullname" . }} -n {{ .Release.Namespace }} 8080:{{ .Values.service.port }}
     # then open http://localhost:8080
{{- end }}
